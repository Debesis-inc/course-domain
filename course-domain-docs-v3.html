src/main/java/…/model/Course.java @Document Spring Data MongoDB
@Document(collection = "course") @AllArgsConstructor @NoArgsConstructor @Data public class Course { @Id // org.springframework.data.annotation.Id private String id; // String ObjectId (was Long in PostgreSQL) private String code; private String title; private String slug; // auto-generated from title in service private String description; private String instructorId; private String level; private String languageCode; private String thumbnailUrl; private Integer durationHours; @CreatedDate // auto-stamped by @EnableMongoAuditing private Instant createdAt; private Instant updatedAt; // set manually in service }
src/main/java/…/repository/CourseRepository.java MongoRepository
public interface CourseRepository extends MongoRepository<Course, String> { // Spring Data MongoDB generates: find({code: {$regex: /^value$/i}}) boolean existsByCodeIgnoreCase(String code); // Used to prevent duplicate instructors per course boolean existsByInstructorIdIgnoreCase(String instructorId); // Prevents duplicate course titles boolean existsByTitleIgnoreCase(String title); }
src/main/java/…/service/impl/CourseServiceImpl.java @Service MapStruct
@Service @AllArgsConstructor @Slf4j public class CourseServiceImpl implements CourseService { private final CourseRepository courseRepository; private final CourseMapper courseMapper; // MapStruct → toResponseDTO() private final CourseConverter courseConverter; // MapStruct → toEntity() @Override public CourseResponseDTO createCourse(CourseRequestDTO dto) { validateUniqueCourseFields(dto); // throws DuplicateCourseFieldException Course course = courseConverter.toEntity(dto); // MapStruct mapping Instant now = Instant.now(); course.setSlug(toSlug(dto.getTitle())); // "Spring Boot" → "spring-boot" course.setCreatedAt(now); course.setUpdatedAt(now); Course saved = courseRepository.save(course); return courseMapper.toResponseDTO(saved); } private void validateUniqueCourseFields(CourseRequestDTO dto) { if (courseRepository.existsByCodeIgnoreCase(dto.getCode())) throw new DuplicateCourseFieldException("code", dto.getCode()); if (courseRepository.existsByInstructorIdIgnoreCase(dto.getInstructorId())) throw new DuplicateCourseFieldException("instructorId", dto.getInstructorId()); if (courseRepository.existsByTitleIgnoreCase(dto.getTitle())) throw new DuplicateCourseFieldException("title", dto.getTitle()); } private String toSlug(String value) { return value.toLowerCase(Locale.ROOT) .replaceAll("[^a-z0-9]+", "-") .replaceAll("(^-|-$)", ""); } }
src/main/java/…/controller/CourseController.java @RestController
@RestController @AllArgsConstructor @RequestMapping("/api/v1/courses") public class CourseController { private final CourseService courseService; @PostMapping public ResponseEntity<CourseResponseDTO> createCourse( @Valid @RequestBody CourseRequestDTO courseRequestDTO) { CourseResponseDTO created = courseService.createCourse(courseRequestDTO); return ResponseEntity.status(HttpStatus.CREATED).body(created); } }
src/main/java/…/dto/request/CourseRequestDTO.java Bean Validation
@AllArgsConstructor @NoArgsConstructor @Data public class CourseRequestDTO { @NotBlank @Size(max = 50) private String code; // e.g. "SB-2026-001" @NotBlank @Size(max = 150) private String title; @Size(max = 5000) private String description; @NotBlank @Pattern(regexp = "^[A-Z]{2}\\d{4}$", message = "instructorId must match format TM1234") private String instructorId; // e.g. "GV0001" — 2 uppercase + 4 digits @Size(max = 50) private String level; // BEGINNER / INTERMEDIATE / ADVANCED @Size(max = 10) private String languageCode; // e.g. "en", "it", "nl" @Size(max = 500) private String thumbnailUrl; @PositiveOrZero private Integer durationHours; }
src/main/java/…/dto/response/CourseResponseDTO.java

Deployment Guide
1
Build Spring Boot JAR
Package locally skipping tests.
./mvnw clean package -DskipTests # → target/course-domain-0.0.1-SNAPSHOT.jar
2
Upload JAR to Droplet
SCP the jar to the app folder on VPS.
scp target/course-domain-0.0.1-SNAPSHOT.jar \ root@165.245.223.137:/opt/course-domain/course-domain.jar
3
Start Spring Boot Service
systemd manages the process and auto-restarts on crash or reboot.
ssh root@165.245.223.137 systemctl restart course-domain systemctl status course-domain journalctl -u course-domain -f
4
Add Missing MapStruct Files
CourseMapper, CourseConverter and DuplicateCourseFieldException are referenced but not yet in the repo. These need to be created.
# Files needed: src/main/java/…/mapper/CourseMapper.java ← @Mapper interface src/main/java/…/mapper/CourseConverter.java ← @Mapper interface src/main/java/…/exception/DuplicateCourseFieldException.java
5
Deploy Frontend
Build Angular/React locally and upload the dist folder.
ng build --configuration production scp -r dist/* root@165.245.223.137:/opt/course-frontend/ # Nginx will serve it on port 80 at /
